using System;
namespace Task3
{
class Program 
{
  static void Main(string[] args) 
     {
       
      int num,count=0,pow=0,i=2;
      num = Convert.ToInt32(Console.ReadLine());
      if(num==0 || num==1)Console.WriteLine("1");
      else{
      while(i<=num){
          i=i<<1;
          count++;
          pow=i;
      }
         if(num+1==pow){count++;}
      Console.WriteLine(count);
      }
        }
     }
    
}
