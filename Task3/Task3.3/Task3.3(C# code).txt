using System;

{
class Program 
{
  static void Main(string[] args) 
     {
       
      int num,counter=0;
      num = Convert.ToInt32(Console.ReadLine());
      for(int i=2;i<num;i++){
          if(num%i==0)counter++;
      }
      if(counter!=0)Console.WriteLine("complex");
      else {Console.WriteLine("simple");}
      }
        }
     }
